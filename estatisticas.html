Como configurar a integração com o Google Sheets?

Como a sua página está hospedada no GitHub Pages (um servidor estático), não podemos usar um banco de dados tradicional. Podemos usar a sua planilha do Google Sheets como banco de dados gratuito. Siga o passo a passo para configurar:

  1. Abra a sua planilha do Google Sheets (a mesma que você já utiliza para os links do Zoom).
  2. No menu superior, acesse Extensões > Apps Script.
  3. Apague todo o código que estiver lá e cole o seguinte código:
function doPost(e) { var rawData = e.postData.contents; var data; try { data = JSON.parse(rawData); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: "Invalid JSON"})) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } var sheetName = "Estatísticas"; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(["Timestamp", "Unidade", "Audiência", "Papel"]); sheet.getRange("A1:D1").setFontWeight("bold").setBackground("#d9e1f2"); } // Converte data ISO ou usa hora local do servidor do Apps Script var dateStr = data.timestamp ? new Date(data.timestamp) : new Date(); sheet.appendRow([ dateStr, data.unidade || "", data.audiencia || "", data.papel || "" ]); return ContentService.createTextOutput(JSON.stringify({status: "success"})) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } function doGet() { var sheetName = "Estatísticas"; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } var values = sheet.getDataRange().getValues(); if (values.length <= 1) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } var headers = values[0]; var list = []; for (var i = 1; i < values.length; i++) { var item = {}; for (var j = 0; j < headers.length; j++) { var key = headers[j].toLowerCase(); var val = values[i][j]; // Se for objeto de data, formata para string ISO if (val instanceof Date) { val = val.toISOString(); } item[key] = val; } list.push(item); } return ContentService.createTextOutput(JSON.stringify(list)) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } function doOptions(e) { return ContentService.createTextOutput("") .addHeader("Access-Control-Allow-Origin", "*") .addHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS") .addHeader("Access-Control-Allow-Headers", "Content-Type"); }
  1. No canto superior direito, clique em Implantar > Nova implantação.
  2. No tipo de implantação (ícone de engrenagem), selecione App da Web.
  3. Configure as opções:
    • Descrição: Registro de Acessos TRT12
    • Executar como: Eu (seu email do Google)
    • Quem tem acesso: Qualquer pessoa (necessário para que os usuários possam salvar os logs).
  4. Clique em Implantar e autorize o acesso à sua planilha se o Google solicitar.
  5. Copie o link do App da Web gerado (URL que termina com `/exec`).
  6. Agora, cole esse link no campo URL de API do Google Apps Script acima neste painel e clique em "Conectar".
  7. (Opcional) Abra o arquivo `trt12-acesso-zoom-google-sheet.html` em seu editor e cole essa URL na constante `STATS_SCRIPT_URL` (próximo à linha 295) para que o envio automático em produção comece a rodar para todos os usuários!
